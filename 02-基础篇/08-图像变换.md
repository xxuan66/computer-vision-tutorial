## 8.1 傅里叶变换

### 8.1.1 原理

**傅里叶变换**将图像从空间域转换到频率域。

**核心思想：**
> 任何周期性信号都可以表示为正弦和余弦函数的叠加

**数学公式：**
```
F(u, v) = ΣΣ f(x, y) × e^(-j2π(ux/M + vy/N))

其中：
- f(x, y): 空间域图像
- F(u, v): 频率域表示
- u, v: 频率变量
```

### 8.1.2 代码实现

```python
import cv2
import numpy as np
import matplotlib.pyplot as plt

# 读取图像（灰度）
img = cv2.imread('image.jpg', cv2.IMREAD_GRAYSCALE)

# 傅里叶变换
dft = cv2.dft(np.float32(img), flags=cv2.DFT_COMPLEX_OUTPUT)
dft_shift = np.fft.fftshift(dft)

# 计算幅度谱
magnitude_spectrum = 20 * np.log(cv2.magnitude(dft_shift[:, :, 0], dft_shift[:, :, 1]))

# 显示
plt.figure(figsize=(12, 6))

plt.subplot(121)
plt.imshow(img, cmap='gray')
plt.title('Input Image')
plt.axis('off')

plt.subplot(122)
plt.imshow(magnitude_spectrum, cmap='gray')
plt.title('Magnitude Spectrum')
plt.axis('off')

plt.tight_layout()
plt.show()
```

### 8.1.3 应用：频域滤波

```python
import cv2
import numpy as np

# 读取图像
img = cv2.imread('image.jpg', cv2.IMREAD_GRAYSCALE)

# 傅里叶变换
dft = cv2.dft(np.float32(img), flags=cv2.DFT_COMPLEX_OUTPUT)
dft_shift = np.fft.fftshift(dft)

# 创建低通滤波器
rows, cols = img.shape
crow, ccol = rows // 2, cols // 2
mask = np.zeros((rows, cols, 2), np.uint8)

# 中心区域保留（低通）
r = 30
mask[crow-r:crow+r, ccol-r:ccol+r] = 1

# 应用滤波器
fshift = dft_shift * mask

# 傅里叶逆变换
f_ishift = np.fft.ifftshift(fshift)
img_back = cv2.idft(f_ishift)
img_back = cv2.magnitude(img_back[:, :, 0], img_back[:, :, 1])

# 显示
cv2.imshow('Original', img)
cv2.imshow('Filtered', img_back)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

## 8.2 霍夫变换

### 8.2.1 直线检测

**霍夫变换**用于检测图像中的几何形状。

```python
import cv2
import numpy as np

# 读取图像
img = cv2.imread('image.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# 边缘检测
edges = cv2.Canny(gray, 50, 150, apertureSize=3)

# 霍夫直线检测
lines = cv2.HoughLines(edges, 1, np.pi / 180, 200)

# 绘制直线
if lines is not None:
    for line in lines:
        rho, theta = line[0]
        a = np.cos(theta)
        b = np.sin(theta)
        x0 = a * rho
        y0 = b * rho
        x1 = int(x0 + 1000 * (-b))
        y1 = int(y0 + 1000 * (a))
        x2 = int(x0 - 1000 * (-b))
        y2 = int(y0 - 1000 * (a))
        cv2.line(img, (x1, y1), (x2, y2), (0, 0, 255), 2)

# 显示
cv2.imshow('Lines', img)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

### 8.2.2 概率霍夫变换

```python
import cv2
import numpy as np

# 读取图像
img = cv2.imread('image.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# 边缘检测
edges = cv2.Canny(gray, 50, 150)

# 概率霍夫变换
lines = cv2.HoughLinesP(edges, 1, np.pi/180, threshold=50, 
                        minLineLength=50, maxLineGap=10)

# 绘制直线
if lines is not None:
    for line in lines:
        x1, y1, x2, y2 = line[0]
        cv2.line(img, (x1, y1), (x2, y2), (0, 255, 0), 2)

# 显示
cv2.imshow('Probabilistic Hough', img)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

### 8.2.3 圆检测

```python
import cv2
import numpy as np

# 读取图像
img = cv2.imread('circles.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# 高斯模糊
gray_blur = cv2.GaussianBlur(gray, (9, 9), 2)

# 霍夫圆检测
circles = cv2.HoughCircles(gray_blur, cv2.HOUGH_GRADIENT, dp=1, minDist=50,
                           param1=50, param2=50, 
                           minRadius=0, maxRadius=0)

# 绘制圆
if circles is not None:
    circles = np.round(circles[0, :]).astype("int")
    for (x, y, r) in circles:
        cv2.circle(img, (x, y), r, (0, 255, 0), 2)
        cv2.circle(img, (x, y), 2, (0, 0, 255), 3)

# 显示
cv2.imshow('Circles', img)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

## 8.3 仿射变换

### 8.3.1 平移

```python
import cv2
import numpy as np

# 读取图像
img = cv2.imread('image.jpg')
rows, cols = img.shape[:2]

# 平移矩阵（向右 50 像素，向下 30 像素）
M = np.float32([[1, 0, 50],
```

## 8.4 透视变换

### 8.4.1 原理

**透视变换**可以改变图像的视角。

```python
import cv2
import numpy as np

# 读取图像
img = cv2.imread('image.jpg')
rows, cols = img.shape[:2]

# 定义源点和目标点
pts1 = np.float32([[56, 65], [238, 52], [28, 387], [386, 390]])
pts2 = np.float32([[0, 0], [300, 0], [0, 300], [300, 300]])

# 计算透视变换矩阵
M = cv2.getPerspectiveTransform(pts1, pts2)

# 应用透视变换
dst = cv2.warpPerspective(img, M, (300, 300))

# 显示
cv2.imshow('Original', img)
cv2.imshow('Perspective', dst)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

### 8.4.2 应用：文档扫描

```python
import cv2
import numpy as np

def scan_document(img, pts):
    """文档扫描"""
    # 定义目标点（A4 纸比例）
    (tl, tr, br, bl) = pts
    widthA = np.sqrt(((br[0] - bl[0]) ** 2) + ((br[1] - bl[1]) ** 2))
    widthB = np.sqrt(((tr[0] - tl[0]) ** 2) + ((tr[1] - tl[1]) ** 2))
    maxWidth = max(int(widthA), int(widthB))
    
    heightA = np.sqrt(((tr[0] - br[0]) ** 2) + ((tr[1] - br[1]) ** 2))
    heightB = np.sqrt(((tl[0] - bl[0]) ** 2) + ((tl[1] - bl[1]) ** 2))
    maxHeight = max(int(heightA), int(heightB))
    
    # 目标点
    dst = np.array([
        [0, 0],
        [maxWidth - 1, 0],
        [maxWidth - 1, maxHeight - 1],
        [0, maxHeight - 1]], dtype="float32")
    
    # 透视变换
    M = cv2.getPerspectiveTransform(pts, dst)
    warped = cv2.warpPerspective(img, M, (maxWidth, maxHeight))
    
    return warped

# 使用
img = cv2.imread('document.jpg')
pts = np.array([[56, 65], [238, 52], [28, 387], [386, 390]], dtype="float32")
scanned = scan_document(img, pts)

cv2.imshow('Original', img)
cv2.imshow('Scanned', scanned)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

## 8.5 重映射

### 8.5.1 原理

**重映射**可以任意改变图像的像素位置。

```python
import cv2
import numpy as np

# 读取图像
img = cv2.imread('image.jpg')
rows, cols = img.shape[:2]

# 创建映射
map_x = np.zeros((rows, cols), np.float32)
map_y = np.zeros((rows, cols), np.float32)

for i in range(rows):
    for j in range(cols):
        map_x[i, j] = j
        map_y[i, j] = i + 10 * np.sin(2 * np.pi * j / 100)

# 应用重映射
dst = cv2.remap(img, map_x, map_y, cv2.INTER_LINEAR)

# 显示
cv2.imshow('Original', img)
cv2.imshow('Remapped', dst)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

### 8.5.2 应用：图像校正

```python
import cv2
import numpy as np

def correct_distortion(img):
    """校正镜头畸变"""
    rows, cols = img.shape[:2]
    
    # 相机参数（示例）
    fx = cols
    fy = rows
    cx = cols / 2
    cy = rows / 2
    
    # 创建映射
    map_x = np.zeros((rows, cols), np.float32)
    map_y = np.zeros((rows, cols), np.float32)
    
    for i in range(rows):
        for j in range(cols):
            # 归一化坐标
            x = (j - cx) / fx
            y = (i - cy) / fy
            
            # 径向畸变校正
            r2 = x*x + y*y
            radial = 1 + 0.1 * r2
            
            # 映射
            map_x[i, j] = fx * x * radial + cx
            map_y[i, j] = fy * y * radial + cy
    
    # 应用校正
    corrected = cv2.remap(img, map_x, map_y, cv2.INTER_LINEAR)
    
    return corrected

# 使用
img = cv2.imread('distorted.jpg')
corrected = correct_distortion(img)

cv2.imshow('Distorted', img)
cv2.imshow('Corrected', corrected)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
