## 7.1 边缘检测基础

### 7.1.1 什么是边缘

**边缘**是图像中灰度值发生突变的地方，通常对应物体的边界。

**边缘类型：**
- **阶跃边缘** — 灰度值突然变化
- **屋顶边缘** — 灰度值先增后减
- **细线边缘** — 细长的线条

### 7.1.2 边缘检测原理

**基于梯度：**
```
梯度幅度：|G| = √(Gx² + Gy²)
梯度方向：θ = arctan(Gy/Gx)

其中：
- Gx: x 方向梯度
- Gy: y 方向梯度
```

## 7.2 Sobel 算子

### 7.2.1 原理

**Sobel 算子**使用两个 3x3 卷积核分别计算 x 和 y 方向的梯度。

**卷积核：**
```
Gx (垂直边缘)        Gy (水平边缘)
┌─────────┐          ┌─────────┐
│-1  0  1 │          │ 1  2  1 │
│-2  0  2 │          │ 0  0  0 │
│-1  0  1 │          │-1 -2 -1 │
└─────────┘          └─────────┘
```

### 7.2.2 代码实现

```python
import cv2
import numpy as np

# 读取图像（灰度）
img = cv2.imread('image.jpg', cv2.IMREAD_GRAYSCALE)

# Sobel 算子
sobelx = cv2.Sobel(img, cv2.CV_64F, 1, 0, ksize=3)
sobely = cv2.Sobel(img, cv2.CV_64F, 0, 1, ksize=3)

# 计算梯度幅度
sobel_magnitude = cv2.magnitude(sobelx, sobely)

# 转换为 uint8
sobelx_abs = cv2.convertScaleAbs(sobelx)
sobely_abs = cv2.convertScaleAbs(sobely)
sobel_magnitude_abs = cv2.convertScaleAbs(sobel_magnitude)

# 显示
cv2.imshow('Original', img)
cv2.imshow('Sobel X', sobelx_abs)
cv2.imshow('Sobel Y', sobely_abs)
cv2.imshow('Magnitude', sobel_magnitude_abs)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

## 7.3 Laplacian 算子

### 7.3.1 原理

**Laplacian 算子**计算图像的二阶导数，对边缘更敏感。

**卷积核：**
```
┌─────────┐
│ 0  1  0 │
│ 1 -4  1 │
│ 0  1  0 │
└─────────┘
```

### 7.3.2 代码实现

```python
import cv2

# 读取图像
img = cv2.imread('image.jpg', cv2.IMREAD_GRAYSCALE)

# Laplacian 算子
laplacian = cv2.Laplacian(img, cv2.CV_64F)
laplacian_abs = cv2.convertScaleAbs(laplacian)

# 显示
cv2.imshow('Original', img)
cv2.imshow('Laplacian', laplacian_abs)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

## 7.4 Canny 边缘检测

### 7.4.1 算法步骤

**Canny 算法**是最常用的边缘检测方法。

**步骤：**
1. 高斯滤波去噪
2. 计算梯度幅度和方向
3. 非极大值抑制
4. 双阈值检测
5. 边缘连接

### 7.4.2 代码实现

```python
import cv2

# 读取图像
img = cv2.imread('image.jpg', cv2.IMREAD_GRAYSCALE)

# Canny 边缘检测
edges = cv2.Canny(img, 50, 150)

# 不同阈值对比
edges1 = cv2.Canny(img, 30, 80)
edges2 = cv2.Canny(img, 100, 200)

# 显示
cv2.imshow('Original', img)
cv2.imshow('Canny 50-150', edges)
cv2.imshow('Canny 30-80', edges1)
cv2.imshow('Canny 100-200', edges2)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

### 7.4.3 自动阈值 Canny

```python
import cv2
import numpy as np

def auto_canny(image, sigma=0.33):
    """自动计算 Canny 阈值"""
    # 计算中值
    v = np.median(image)
    
    # 自动阈值
    lower = int(max(0, (1.0 - sigma) * v))
    upper = int(min(255, (1.0 + sigma) * v))
    
    # 应用 Canny
    edges = cv2.Canny(image, lower, upper)
    
    return edges

# 使用
img = cv2.imread('image.jpg', cv2.IMREAD_GRAYSCALE)
edges = auto_canny(img)

cv2.imshow('Original', img)
cv2.imshow('Auto Canny', edges)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

## 7.5 角点检测

### 7.5.1 Harris 角点检测

```python
import cv2
import numpy as np

# 读取图像
img = cv2.imread('image.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# Harris 角点检测
dst = cv2.cornerHarris(gray, 2, 3, 0.04)

# 阈值
dst_dilated = cv2.dilate(dst, None)
img[dst_dilated > 0.01 * dst_dilated.max()] = [0, 0, 255]

# 显示
cv2.imshow('Harris Corners', img)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

### 7.5.2 Shi-Tomasi 角点检测

```python
import cv2
import numpy as np

# 读取图像
img = cv2.imread('image.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# Shi-Tomasi 角点检测
corners = cv2.goodFeaturesToTrack(gray, 50, 0.01, 10)
corners = np.int0(corners)

# 绘制角点
for corner in corners:
    x, y = corner.ravel()
    cv2.circle(img, (x, y), 5, (0, 255, 0), -1)

# 显示
cv2.imshow('Shi-Tomasi Corners', img)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

## 7.6 SIFT 特征

### 7.6.1 原理

**SIFT**（Scale-Invariant Feature Transform）是尺度不变特征变换。

**特点：**
- ✅ 尺度不变性
- ✅ 旋转不变性
- ✅ 光照不变性
- ❌ 计算量大

### 7.6.2 代码实现

```python
import cv2

# 读取图像
img = cv2.imread('image.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# 创建 SIFT 对象
sift = cv2.SIFT_create()

# 检测关键点
keypoints = sift.detect(gray, None)

# 计算描述子
keypoints, descriptors = sift.compute(gray, keypoints)

# 绘制关键点
img_with_keypoints = cv2.drawKeypoints(gray, keypoints, None, flags=cv2.DRAW_MATCHES_FLAGS_DRAW_RICH_KEYPOINTS)

# 显示
cv2.imshow('SIFT Keypoints', img_with_keypoints)
cv2.waitKey(0)
cv2.destroyAllWindows()

print(f"检测到 {len(keypoints)} 个关键点")
print(f"描述子形状：{descriptors.shape}")
```

## 7.7 SURF 特征

### 7.7.1 原理

**SURF**（Speeded-Up Robust Features）是加速的鲁棒特征。

**vs SIFT：**
- ✅ 速度更快
- ✅ 性能相当
- ❌ 专利限制

### 7.7.2 代码实现

```python
import cv2

# 读取图像
img = cv2.imread('image.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# 创建 SURF 对象
surf = cv2.SURF_create(400)

# 检测关键点和描述子
keypoints, descriptors = surf.detectAndCompute(gray, None)

# 绘制关键点
img_with_keypoints = cv2.drawKeypoints(gray, keypoints, None, (255, 0, 0), 2)

# 显示
cv2.imshow('SURF Keypoints', img_with_keypoints)
cv2.waitKey(0)
cv2.destroyAllWindows()

print(f"检测到 {len(keypoints)} 个关键点")
```

## 7.8 ORB 特征

### 7.8.1 原理

**ORB**（Oriented FAST and Rotated BRIEF）是免费的快速特征。

**vs SIFT/SURF：**
- ✅ 免费开源
- ✅ 速度最快
- ✅ 实时应用
- ❌ 性能稍差

### 7.8.2 代码实现

```python
import cv2

# 读取图像
img = cv2.imread('image.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# 创建 ORB 对象
orb = cv2.ORB_create(nfeatures=500)

# 检测关键点和描述子
keypoints, descriptors = orb.detectAndCompute(gray, None)

# 绘制关键点
img_with_keypoints = cv2.drawKeypoints(gray, keypoints, None, (0, 255, 0), 2)

# 显示
cv2.imshow('ORB Keypoints', img_with_keypoints)
cv2.waitKey(0)
cv2.destroyAllWindows()

print(f"检测到 {len(keypoints)} 个关键点")
```

## 7.9 特征匹配

### 7.9.1 特征匹配

```python
import cv2

# 读取图像
img1 = cv2.imread('image1.jpg', cv2.IMREAD_GRAYSCALE)
img2 = cv2.imread('image2.jpg', cv2.IMREAD_GRAYSCALE)

# 创建 ORB 对象
orb = cv2.ORB_create()

# 检测关键点和描述子
kp1, des1 = orb.detectAndCompute(img1, None)
kp2, des2 = orb.detectAndCompute(img2, None)

# 特征匹配
bf = cv2.BFMatcher(cv2.NORM_HAMMING, crossCheck=True)
matches = bf.match(des1, des2)

# 按距离排序
matches = sorted(matches, key=lambda x: x.distance)

# 绘制匹配结果
img_matches = cv2.drawMatches(img1, kp1, img2, kp2, matches[:50], None, flags=2)

# 显示
cv2.imshow('Matches', img_matches)
cv2.waitKey(0)
cv2.destroyAllWindows()

print(f"匹配到 {len(matches)} 个特征点")
```

### 7.9.2 应用：图像拼接

```python
import cv2
import numpy as np

def image_stitching(img1, img2):
    """图像拼接"""
    # 转换为灰度
    gray1 = cv2.cvtColor(img1, cv2.COLOR_BGR2GRAY)
    gray2 = cv2.cvtColor(img2, cv2.COLOR_BGR2GRAY)
    
    # 检测特征
    orb = cv2.ORB_create()
    kp1, des1 = orb.detectAndCompute(gray1, None)
    kp2, des2 = orb.detectAndCompute(gray2, None)
    
    # 特征匹配
    bf = cv2.BFMatcher(cv2.NORM_HAMMING)
    matches = bf.knnMatch(des1, des2, k=2)
    
    # 筛选好的匹配
    good_matches = []
    for m, n in matches:
        if m.distance < 0.75 * n.distance:
            good_matches.append(m)
    
    # 计算单应性矩阵
    if len(good_matches) > 4:
        pts1 = np.float32([kp1[m.queryIdx].pt for m in good_matches]).reshape(-1, 1, 2)
        pts2 = np.float32([kp2[m.trainIdx].pt for m in good_matches]).reshape(-1, 1, 2)
        
        H, mask = cv2.findHomography(pts1, pts2, cv2.RANSAC, 5.0)
        
        # 图像拼接
        result = cv2.warpPerspective(img1, H, (img1.shape[1] + img2.shape[1], img1.shape[0]))
        result[0:img2.shape[0], 0:img2.shape[1]] = img2
        
        return result
    
    return None

# 使用
img1 = cv2.imread('left.jpg')
img2 = cv2.imread('right.jpg')
result = image_stitching(img1, img2)

if result is not None:
    cv2.imshow('Stitched Image', result)
    cv2.waitKey(0)
    cv2.destroyAllWindows()
```
