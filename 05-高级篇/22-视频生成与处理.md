## 22.2 视频分类

### 22.2.1 3D CNN

**3D CNN**同时处理空间和时间维度。

```python
import torch
import torch.nn as nn

class VideoClassifier3D(nn.Module):
    """3D CNN 视频分类"""
    
    def __init__(self, num_classes=10):
        super().__init__()
        
        self.features = nn.Sequential(
            nn.Conv3d(3, 64, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.MaxPool3d(2),
            
            nn.Conv3d(64, 128, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.MaxPool3d(2)
        )
        
        self.classifier = nn.Sequential(
            nn.AdaptiveAvgPool3d(1),
            nn.Flatten(),
            nn.Linear(128, num_classes)
        )
    
    def forward(self, x):
        # x: [batch, channels, frames, height, width]
        x = self.features(x)
        x = self.classifier(x)
        return x
```

## 22.6 实战：视频分析系统

### 22.6.1 系统架构

```
视频输入 → 帧提取 → 目标检测 → 跟踪 → 行为分析 → 结果输出
```

### 22.6.2 完整代码

```python
import cv2
import torch
from sort_tracker import Sort

class VideoAnalysisSystem:
    """视频分析系统"""
    
    def __init__(self):
        # 加载检测模型
        self.detector = torch.hub.load('ultralytics/yolov5', 'yolov5s')
        
        # 加载跟踪器
        self.tracker = Sort()
    
    def analyze(self, video_path, output_path):
        """分析视频"""
        cap = cv2.VideoCapture(video_path)
        
        fps = cap.get(cv2.CAP_PROP_FPS)
        width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
        height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
        
        fourcc = cv2.VideoWriter_fourcc(*'mp4v')
        out = cv2.VideoWriter(output_path, fourcc, fps, (width, height))
        
        while cap.isOpened():
            ret, frame = cap.read()
            if not ret:
                break
            
            # 检测
            results = self.detector(frame)
            detections = results.pandas().xyxy[0]
            
            # 跟踪
            tracked_objects = self.tracker.update(detections)
            
            # 绘制
            for track in tracked_objects:
                x1, y1, x2, y2, track_id = track
                cv2.rectangle(frame, (int(x1), int(y1)), (int(x2), int(y2)), (0, 255, 0), 2)
                cv2.putText(frame, f'ID: {track_id}', (int(x1), int(y1)-10),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
            
            out.write(frame)
        
        cap.release()
        out.release()
        print(f"分析完成：{output_path}")

# 使用
system = VideoAnalysisSystem()
system.analyze('input.mp4', 'output.mp4')
```
