## 17.2 YOLOv5 使用

### 17.2.1 安装

```bash
# 克隆 YOLOv5 仓库
git clone https://github.com/ultralytics/yolov5
cd yolov5

# 安装依赖
pip install -r requirements.txt
```

### 17.2.2 图像检测

```python
import torch

# 加载模型
model = torch.hub.load('ultralytics/yolov5', 'yolov5s')

# 检测
img = 'image.jpg'
results = model(img)

# 显示结果
results.show()

# 获取检测结果
df = results.pandas().xyxy[0]
print(df)
```

### 17.2.3 视频检测

```python
import torch
import cv2

# 加载模型
model = torch.hub.load('ultralytics/yolov5', 'yolov5s')

# 打开摄像头
cap = cv2.VideoCapture(0)

while True:
    ret, frame = cap.read()
    
    # 检测
    results = model(frame)
    
    # 绘制结果
    annotated_frame = results.render()[0]
    
    # 显示
    cv2.imshow('YOLOv5 Detection', annotated_frame)
    
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```

## 17.3 Web 应用

### 17.3.1 Streamlit 应用

```python
# app.py
import streamlit as st
import torch
from PIL import Image
import cv2
import numpy as np

# 页面配置
st.set_page_config(page_title="目标检测系统", layout="wide")

# 标题
st.title("🎯 实时目标检测系统")

# 侧边栏
st.sidebar.header("设置")
model_type = st.sidebar.selectbox("模型", ["yolov5s", "yolov5m", "yolov5l"])
confidence = st.sidebar.slider("置信度阈值", 0.0, 1.0, 0.25)

# 加载模型
@st.cache_resource
def load_model(model_type):
    return torch.hub.load('ultralytics/yolov5', model_type)

model = load_model(model_type)

# 上传文件
uploaded_file = st.file_uploader("上传图片", type=["jpg", "jpeg", "png"])

if uploaded_file:
    # 显示原图
    col1, col2 = st.columns(2)
    
    with col1:
        st.subheader("原图")
        image = Image.open(uploaded_file)
        st.image(image, use_column_width=True)
    
    # 检测
    with col2:
        st.subheader("检测结果")
        results = model(image)
        st.image(results.render()[0], use_column_width=True)
    
    # 显示检测结果
    st.subheader("检测详情")
    df = results.pandas().xyxy[0]
    if len(df) > 0:
        st.dataframe(df)
    else:
        st.info("未检测到物体")

# 摄像头检测
st.subheader("实时摄像头检测")
if st.button("启动摄像头"):
    cap = cv2.VideoCapture(0)
    st_frame = st.empty()
    
    while cap.isOpened():
        ret, frame = cap.read()
        if not ret:
            break
        
        # 检测
        results = model(frame)
        annotated_frame = results.render()[0]
        
        # 显示
        st_frame.image(annotated_frame, channels="BGR", use_column_width=True)
        
        if st.button("停止"):
            break
    
    cap.release()
```

## 17.5 模型优化

### 17.5.1 导出 ONNX

```python
import torch

# 加载模型
model = torch.hub.load('ultralytics/yolov5', 'yolov5s')

# 导出 ONNX
torch.onnx.export(
    model,
    torch.zeros(1, 3, 640, 640),
    'yolov5s.onnx',
    opset_version=12,
    input_names=['images'],
    output_names=['output']
)
```

### 17.5.2 ONNX Runtime 推理

```python
import onnxruntime
import cv2
import numpy as np

# 加载模型
session = onnxruntime.InferenceSession('yolov5s.onnx')

# 预处理
img = cv2.imread('image.jpg')
img = cv2.resize(img, (640, 640))
img = img.astype(np.float32) / 255.0
img = np.transpose(img, (2, 0, 1))
img = np.expand_dims(img, axis=0)

# 推理
outputs = session.run(None, {'images': img})

print(f"输出形状：{outputs[0].shape}")
```
