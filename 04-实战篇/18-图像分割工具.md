## 18.2 U-Net 实现

### 18.2.1 模型定义

```python
import torch
import torch.nn as nn

class UNet(nn.Module):
    """U-Net 实现"""
    
    def __init__(self, n_channels=1, n_classes=2):
        super().__init__()
        
        # 编码器
        self.enc1 = self._block(n_channels, 64)
        self.enc2 = self._block(64, 128)
        self.enc3 = self._block(128, 256)
        self.enc4 = self._block(256, 512)
        
        # 瓶颈
        self.bottleneck = self._block(512, 1024)
        
        # 解码器
        self.upconv4 = nn.ConvTranspose2d(1024, 512, 2, stride=2)
        self.dec4 = self._block(1024, 512)
        
        self.upconv3 = nn.ConvTranspose2d(512, 256, 2, stride=2)
        self.dec3 = self._block(512, 256)
        
        self.upconv2 = nn.ConvTranspose2d(256, 128, 2, stride=2)
        self.dec2 = self._block(256, 128)
        
        self.upconv1 = nn.ConvTranspose2d(128, 64, 2, stride=2)
        self.dec1 = self._block(128, 64)
        
        # 输出
        self.out_conv = nn.Conv2d(64, n_classes, 1)
    
    def _block(self, in_channels, out_channels):
        return nn.Sequential(
            nn.Conv2d(in_channels, out_channels, 3, padding=1),
            nn.ReLU(inplace=True),
            nn.Conv2d(out_channels, out_channels, 3, padding=1),
            nn.ReLU(inplace=True)
        )
    
    def forward(self, x):
        # 编码器
        enc1 = self.enc1(x)
        enc2 = self.enc2(nn.MaxPool2d(2)(enc1))
        enc3 = self.enc3(nn.MaxPool2d(2)(enc2))
        enc4 = self.enc4(nn.MaxPool2d(2)(enc3))
        
        # 瓶颈
        bottleneck = self.bottleneck(nn.MaxPool2d(2)(enc4))
        
        # 解码器
        dec4 = self.upconv4(bottleneck)
        dec4 = torch.cat((dec4, enc4), dim=1)
        dec4 = self.dec4(dec4)
        
        dec3 = self.upconv3(dec4)
        dec3 = torch.cat((dec3, enc3), dim=1)
        dec3 = self.dec3(dec3)
        
        dec2 = self.upconv2(dec3)
        dec2 = torch.cat((dec2, enc2), dim=1)
        dec2 = self.dec2(dec2)
        
        dec1 = self.upconv1(dec2)
        dec1 = torch.cat((dec1, enc1), dim=1)
        dec1 = self.dec1(dec1)
        
        return self.out_conv(dec1)
```

## 18.3 数据处理

### 18.3.1 DICOM 读取

```python
import SimpleITK as sitk
import numpy as np

def load_dicom(path):
    """加载 DICOM 图像"""
    reader = sitk.ImageFileReader()
    reader.SetFileName(path)
    image = reader.Execute()
    
    # 转换为 numpy 数组
    image_array = sitk.GetArrayFromImage(image)
    
    return image_array, image

def save_dicom(image_array, reference_image, output_path):
    """保存为 DICOM"""
    image = sitk.GetImageFromArray(image_array)
    image.CopyInformation(reference_image)
    
    writer = sitk.ImageFileWriter()
    writer.SetFileName(output_path)
    writer.Execute(image)
```

## 18.4 训练与评估

### 18.4.1 损失函数

```python
import torch
import torch.nn as nn

class DiceLoss(nn.Module):
    """Dice 损失"""
    
    def forward(self, pred, target):
        smooth = 1e-5
        pred = torch.sigmoid(pred)
        intersection = pred * target
        dice = (2 * intersection.sum() + smooth) / (pred.sum() + target.sum() + smooth)
        return 1 - dice

class CombinedLoss(nn.Module):
    """BCE + Dice 组合损失"""
    
    def __init__(self):
        super().__init__()
        self.bce = nn.BCEWithLogitsLoss()
        self.dice = DiceLoss()
    
    def forward(self, pred, target):
        return self.bce(pred, target) + self.dice(pred, target)
```

### 18.4.2 训练循环

```python
def train(model, dataloader, criterion, optimizer, device):
    model.train()
    running_loss = 0.0
    
    for images, masks in dataloader:
        images = images.to(device)
        masks = masks.to(device)
        
        outputs = model(images)
        loss = criterion(outputs, masks)
        
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        
        running_loss += loss.item()
    
    return running_loss / len(dataloader)
```

## 18.5 Web 应用

### 18.5.1 Gradio 界面

```python
import gradio as gr
import torch
from PIL import Image
import numpy as np

# 加载模型
model = UNet(n_channels=1, n_classes=2)
model.load_state_dict(torch.load('unet_best.pth'))
model.eval()

def segment_image(image):
    """分割图像"""
    # 预处理
    image_np = np.array(image)
    image_tensor = torch.from_numpy(image_np).float() / 255.0
    image_tensor = image_tensor.unsqueeze(0).unsqueeze(0)
    
    # 推理
    with torch.no_grad():
        output = model(image_tensor)
        mask = torch.sigmoid(output).squeeze().numpy()
    
    # 后处理
    mask = (mask > 0.5).astype(np.uint8) * 255
    mask_pil = Image.fromarray(mask)
    
    return image, mask_pil

# 创建界面
iface = gr.Interface(
    fn=segment_image,
    inputs=gr.Image(type="pil"),
    outputs=[gr.Image(type="pil"), gr.Image(type="pil")],
    title="医学图像分割工具",
    description="上传医学图像，自动分割目标区域"
)

iface.launch()
```

### 18.6.1 Volume Rendering

import plotly.graph_objects as go
import numpy as np

def visualize_3d(volume, mask):
    """3D 可视化"""
    fig = go.Figure()
    
    # 原始体积
    fig.add_trace(go.Volume(
        x=np.arange(volume.shape[2]),
        y=np.arange(volume.shape[1]),
        z=np.arange(volume.shape[0]),
        value=volume.flatten(),
        isomin=0.1,
        isomax=0.8,
        opacity=0.1,
        surface_count=25,
        colorscale='Grays'
    ))
    
    # 分割掩码
    fig.add_trace(go.Volume(
        x=np.arange(mask.shape[2]),
        y=np.arange(mask.shape[1]),
        z=np.arange(mask.shape[0]),
        value=mask.flatten(),
        isomin=0.5,
        isomax=1.0,
        opacity=0.5,
