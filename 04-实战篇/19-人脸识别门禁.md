### 19.3.2 使用 MTCNN

```python
from mtcnn import MTCNN
import cv2

# 加载检测器
detector = MTCNN()

# 检测人脸
img = cv2.imread('image.jpg')
```

## 19.5 人脸数据库

### 19.5.1 数据库设计

```python
import sqlite3
import numpy as np

class FaceDatabase:
    """人脸数据库"""
    
    def __init__(self, db_path='faces.db'):
        self.conn = sqlite3.connect(db_path)
        self.create_table()
    
    def create_table(self):
        cursor = self.conn.cursor()
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS faces (
                id INTEGER PRIMARY KEY,
                name TEXT,
                embedding BLOB,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        ''')
        self.conn.commit()
    
    def add_face(self, name, embedding):
        cursor = self.conn.cursor()
        embedding_bytes = embedding.tobytes()
        cursor.execute(
            'INSERT INTO faces (name, embedding) VALUES (?, ?)',
            (name, embedding_bytes)
        )
        self.conn.commit()
    
    def get_all_faces(self):
        cursor = self.conn.cursor()
        cursor.execute('SELECT name, embedding FROM faces')
        rows = cursor.fetchall()
        
        faces = []
        for name, embedding_bytes in rows:
            embedding = np.frombuffer(embedding_bytes, dtype=np.float64)
            faces.append({'name': name, 'embedding': embedding})
        
        return faces
    
    def close(self):
        self.conn.close()
```

## 19.6 人脸比对

### 19.6.1 相似度计算

```python
import numpy as np
from scipy.spatial.distance import euclidean

def compare_faces(embedding1, embedding2, threshold=0.6):
    """人脸比对"""
    distance = euclidean(embedding1, embedding2)
    
    if distance < threshold:
        return True, distance
    else:
        return False, distance
```

### 19.6.2 识别流程

```python
def recognize_face(face_embedding, database, threshold=0.6):
    """识别人脸"""
    faces = database.get_all_faces()
    
    best_match = None
    best_distance = float('inf')
    
    for face in faces:
        distance = euclidean(face_embedding, face['embedding'])
        if distance < best_distance:
            best_distance = distance
            best_match = face['name']
    
    if best_distance < threshold:
        return best_match, best_distance
    else:
        return 'Unknown', best_distance
```

## 19.7 完整系统

### 19.7.1 主程序

```python
import cv2
import dlib
import numpy as np
from datetime import datetime

class AccessControlSystem:
    """门禁系统"""
    
    def __init__(self):
        self.detector = dlib.get_frontal_face_detector()
        self.predictor = dlib.shape_predictor('shape_predictor_68_face_landmarks.dat')
        self.facerec = dlib.face_recognition_model_v1('dlib_face_recognition_resnet_model_v1.dat')
        self.database = FaceDatabase()
    
    def register(self, name, image_path):
        """注册人脸"""
        img = cv2.imread(image_path)
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        faces = self.detector(gray)
        
        if len(faces) == 0:
            return False
        
        face = faces[0]
        shape = self.predictor(gray, face)
        face_descriptor = self.facerec.compute_face_descriptor(img, shape)
        
        self.database.add_face(name, np.array(face_descriptor))
        return True
    
    def verify(self, image_path, threshold=0.6):
        """验证人脸"""
        img = cv2.imread(image_path)
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        faces = self.detector(gray)
        
        if len(faces) == 0:
            return None, None
        
        face = faces[0]
        shape = self.predictor(gray, face)
        face_descriptor = self.facerec.compute_face_descriptor(img, shape)
        
        return recognize_face(np.array(face_descriptor), self.database, threshold)
    
    def run(self):
        """运行门禁系统"""
        cap = cv2.VideoCapture(0)
        
        while True:
            ret, frame = cap.read()
            gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
            faces = self.detector(gray)
            
            for face in faces:
                shape = self.predictor(gray, face)
                face_descriptor = self.facerec.compute_face_descriptor(frame, shape)
                
                name, distance = recognize_face(np.array(face_descriptor), self.database)
                
                # 绘制结果
                x1, y1 = face.left(), face.top()
                x2, y2 = face.right(), face.bottom()
                
                if name != 'Unknown':
                    color = (0, 255, 0)
                    label = f"{name} ({distance:.2f})"
                    # 开门逻辑
                    print(f"✓ {name} 欢迎！门禁已打开")
                else:
                    color = (0, 0, 255)
                    label = "Unknown"
                
                cv2.rectangle(frame, (x1, y1), (x2, y2), color, 2)
                cv2.putText(frame, label, (x1, y1-10), cv2.FONT_HERSHEY_SIMPLEX, 0.9, color, 2)
            
            # 显示
            cv2.imshow('Access Control', frame)
            
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
        
        cap.release()
        cv2.destroyAllWindows()

# 使用
system = AccessControlSystem()
system.register('Alice', 'alice.jpg')
system.run()
```
