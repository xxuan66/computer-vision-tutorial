## 13.1 图像分割基础

### 13.1.1 什么是图像分割

**图像分割**将图像划分为多个区域，每个像素都有一个类别标签。

**vs 目标检测：**
- 目标检测：边界框（矩形）
- 图像分割：像素级掩码（精确形状）

### 13.1.2 分割类型

| 类型 | 说明 | 输出 |
|------|------|------|
| **语义分割** | 像素分类 | 类别掩码 |
| **实例分割** | 区分不同实例 | 实例掩码 |
| **全景分割** | 语义 + 实例 | 统一输出 |

## 13.2 FCN（全卷积网络）

### 13.2.1 原理

**Fully Convolutional Networks** - 第一个端到端的语义分割网络。

**改进：**
- 全连接层 → 卷积层
- 上采样恢复分辨率
- 跳跃连接融合特征

### 13.2.2 使用 FCN

```python
import torch
import torchvision.models as models

# 加载预训练 FCN
model = models.segmentation.fcn_resnet50(pretrained=True)
model.eval()

# 分割
img = torch.rand(1, 3, 224, 224)
with torch.no_grad():
    output = model(img)['out']

print(f"输出形状：{output.shape}")
```

## 13.3 U-Net

### 13.3.1 原理

**U-Net** - 医学图像分割经典网络。

**特点：**
- 编码器 - 解码器结构
- 跳跃连接
- 小数据集效果好

### 13.3.2 实现 U-Net

```python
import torch
import torch.nn as nn

class UNet(nn.Module):
    """U-Net 实现"""
    
    def __init__(self, n_classes=2):
        super().__init__()
        
        # 编码器
        self.enc1 = self._block(3, 64)
        self.enc2 = self._block(64, 128)
        self.enc3 = self._block(128, 256)
        self.enc4 = self._block(256, 512)
        
        # 瓶颈
        self.bottleneck = self._block(512, 1024)
        
        # 解码器
        self.upconv4 = nn.ConvTranspose2d(1024, 512, 2, stride=2)
        self.dec4 = self._block(1024, 512)
        
        self.upconv3 = nn.ConvTranspose2d(512, 256, 2, stride=2)
        self.dec3 = self._block(512, 256)
        
        self.upconv2 = nn.ConvTranspose2d(256, 128, 2, stride=2)
        self.dec2 = self._block(256, 128)
        
        self.upconv1 = nn.ConvTranspose2d(128, 64, 2, stride=2)
        self.dec1 = self._block(128, 64)
        
        # 输出
        self.out_conv = nn.Conv2d(64, n_classes, 1)
    
    def _block(self, in_channels, out_channels):
        return nn.Sequential(
            nn.Conv2d(in_channels, out_channels, 3, padding=1),
            nn.ReLU(inplace=True),
            nn.Conv2d(out_channels, out_channels, 3, padding=1),
            nn.ReLU(inplace=True)
        )
    
    def forward(self, x):
        # 编码器
        enc1 = self.enc1(x)
        enc2 = self.enc2(nn.MaxPool2d(2)(enc1))
        enc3 = self.enc3(nn.MaxPool2d(2)(enc2))
        enc4 = self.enc4(nn.MaxPool2d(2)(enc3))
        
        # 瓶颈
        bottleneck = self.bottleneck(nn.MaxPool2d(2)(enc4))
        
        # 解码器
        dec4 = self.upconv4(bottleneck)
        dec4 = torch.cat((dec4, enc4), dim=1)
        dec4 = self.dec4(dec4)
        
        dec3 = self.upconv3(dec4)
        dec3 = torch.cat((dec3, enc3), dim=1)
        dec3 = self.dec3(dec3)
        
        dec2 = self.upconv2(dec3)
        dec2 = torch.cat((dec2, enc2), dim=1)
        dec2 = self.dec2(dec2)
        
        dec1 = self.upconv1(dec2)
        dec1 = torch.cat((dec1, enc1), dim=1)
        dec1 = self.dec1(dec1)
        
        return self.out_conv(dec1)

# 使用
model = UNet(n_classes=2)
input_tensor = torch.randn(1, 3, 512, 512)
output = model(input_tensor)
print(f"输出形状：{output.shape}")
```

## 13.6 分割评估指标

### 13.6.1 IoU（交并比）

```python
import numpy as np

def calculate_iou(pred, gt):
    """计算 IoU"""
    intersection = np.logical_and(pred, gt)
    union = np.logical_or(pred, gt)
    iou = np.sum(intersection) / np.sum(union)
    return iou

# 使用
pred = np.array([1, 1, 0, 1, 0, 1])
gt = np.array([1, 1, 0, 0, 0, 1])
iou = calculate_iou(pred, gt)
print(f"IoU: {iou:.4f}")
```

### 13.6.2 Dice 系数

```python
def calculate_dice(pred, gt):
    """计算 Dice 系数"""
    intersection = np.logical_and(pred, gt)
    dice = 2 * np.sum(intersection) / (np.sum(pred) + np.sum(gt))
    return dice

# 使用
dice = calculate_dice(pred, gt)
print(f"Dice: {dice:.4f}")
```

### 13.6.3 mIoU（平均 IoU）

```python
def calculate_miou(pred, gt, num_classes):
    """计算 mIoU"""
    ious = []
    for cls in range(num_classes):
        pred_cls = (pred == cls)
        gt_cls = (gt == cls)
        iou = calculate_iou(pred_cls, gt_cls)
        ious.append(iou)
    return np.mean(ious)

# 使用
miou = calculate_miou(pred, gt, num_classes=2)
print(f"mIoU: {miou:.4f}")
```

## 13.7 实战：医学图像分割

### 13.7.1 数据准备

```python
import torch
from torch.utils.data import Dataset
import cv2
import numpy as np

class MedicalSegmentationDataset(Dataset):
    """医学图像分割数据集"""
    
    def __init__(self, img_paths, mask_paths, transform=None):
        self.img_paths = img_paths
        self.mask_paths = mask_paths
        self.transform = transform
    
    def __len__(self):
        return len(self.img_paths)
    
    def __getitem__(self, idx):
        # 加载图像
        img = cv2.imread(self.img_paths[idx], cv2.IMREAD_GRAYSCALE)
        mask = cv2.imread(self.mask_paths[idx], cv2.IMREAD_GRAYSCALE)
        
        # 归一化
        img = img.astype(np.float32) / 255.0
        mask = mask.astype(np.float32) / 255.0
        
        # 添加通道维度
        img = np.expand_dims(img, axis=0)
        mask = np.expand_dims(mask, axis=0)
        
        if self.transform:
            img = self.transform(img)
            mask = self.transform(mask)
        
        return torch.from_numpy(img), torch.from_numpy(mask)
```

### 13.7.2 训练配置

```python
import torch
import torch.nn as nn

# 损失函数
class DiceLoss(nn.Module):
    """Dice 损失"""
    
    def forward(self, pred, target):
        smooth = 1e-5
        pred = torch.sigmoid(pred)
        intersection = pred * target
        dice = (2 * intersection.sum() + smooth) / (pred.sum() + target.sum() + smooth)
        return 1 - dice
```
