img = cv2.imread('image.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

predictor = dlib.shape_predictor('shape_predictor_68_face_landmarks.dat')

### 15.3.2 人脸对齐

```python
import cv2
import numpy as np
import dlib

def align_face(img, landmarks):
    """人脸对齐"""
    # 获取眼睛关键点
    left_eye = np.array([landmarks.part(36).x, landmarks.part(36).y])
    right_eye = np.array([landmarks.part(45).x, landmarks.part(45).y])
    
    # 计算旋转角度
    dy = right_eye[1] - left_eye[1]
    dx = right_eye[0] - left_eye[0]
    angle = np.degrees(np.arctan2(dy, dx))
    
    # 旋转图像
    center = (img.shape[1] // 2, img.shape[0] // 2)
    M = cv2.getRotationMatrix2D(center, angle, 1.0)
    aligned = cv2.warpAffine(img, M, (img.shape[1], img.shape[0]))
    
    return aligned

# 使用
detector = dlib.get_frontal_face_detector()
predictor = dlib.shape_predictor('shape_predictor_68_face_landmarks.dat')

img = cv2.imread('image.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
faces = detector(gray)

for face in faces:
    landmarks = predictor(gray, face)
    aligned = align_face(img, landmarks)
    
    cv2.imshow('Aligned', aligned)
    cv2.waitKey(0)
    cv2.destroyAllWindows()
```

## 15.5 人脸比对

### 15.5.1 相似度计算

```python
import numpy as np
from scipy.spatial.distance import euclidean
from scipy.spatial.distance import cosine

def calculate_similarity(embedding1, embedding2):
    """计算人脸相似度"""
    # 欧氏距离
    distance = euclidean(embedding1, embedding2)
    
    # 余弦相似度
    cos_sim = 1 - cosine(embedding1, embedding2)
    
    return distance, cos_sim

# 使用
embedding1 = np.random.rand(512)
embedding2 = np.random.rand(512)

distance, cos_sim = calculate_similarity(embedding1, embedding2)
print(f"欧氏距离：{distance:.4f}")
print(f"余弦相似度：{cos_sim:.4f}")
```

### 15.5.2 人脸识别系统

```python
import cv2
import numpy as np
import dlib
from scipy.spatial.distance import euclidean

class FaceRecognitionSystem:
    """人脸识别系统"""
    
    def __init__(self):
        self.detector = dlib.get_frontal_face_detector()
        self.predictor = dlib.shape_predictor('shape_predictor_68_face_landmarks.dat')
        self.facerec = dlib.face_recognition_model_v1('dlib_face_recognition_resnet_model_v1.dat')
        self.known_faces = {}
    
    def register_face(self, name, image_path):
        """注册人脸"""
        img = cv2.imread(image_path)
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        
        faces = self.detector(gray)
        if len(faces) == 0:
            print("未检测到人脸")
            return
        
        face = faces[0]
        shape = self.predictor(gray, face)
        face_descriptor = self.facerec.compute_face_descriptor(img, shape)
        
        self.known_faces[name] = np.array(face_descriptor)
        print(f"✓ {name} 注册成功")
    
    def recognize(self, image_path, threshold=0.6):
        """识别人脸"""
        img = cv2.imread(image_path)
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        
        faces = self.detector(gray)
        results = []
        
        for face in faces:
            shape = self.predictor(gray, face)
            face_descriptor = self.facerec.compute_face_descriptor(img, shape)
            face_embedding = np.array(face_descriptor)
            
            # 比对
            best_match = None
            best_distance = float('inf')
            
            for name, known_embedding in self.known_faces.items():
                distance = euclidean(face_embedding, known_embedding)
                if distance < best_distance:
                    best_distance = distance
                    best_match = name
            
            # 判断
            if best_distance < threshold:
                results.append((best_match, best_distance))
            else:
                results.append(('Unknown', best_distance))
        
        return results

# 使用
system = FaceRecognitionSystem()
system.register_face('Alice', 'alice.jpg')
system.register_face('Bob', 'bob.jpg')

results = system.recognize('test.jpg')
for name, distance in results:
    print(f"识别结果：{name}, 距离：{distance:.4f}")
```
